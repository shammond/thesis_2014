# stest 4 - Line speed change identity test.
# =======
#

sp_create localhost 16001 # Start a session with spd
sp_open ttyS0             # Open a line to sptarget
sp_getport

sp_echo M 100 500         # Ask sptarget to echo its state
sp_put L9600,8,N,1;       # Tell sptarget to set the line to its normal state
pause 3000                # the L command is currently very slow
sp_setport 9600 8 N 1     # Now set this end to the same state
sp_echo M 100 500         # Ask sptarget to echo its state

sp_close FALSE            # Release the line
sp_delete                 # End the session
